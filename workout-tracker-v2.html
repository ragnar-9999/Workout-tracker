function Sparkline({data, color}) {

  if(!data || data.length<2) return null;

  const vals = data.map(d=>d.weight).filter(v=>v>0);

  if(vals.length<2) return null;

  const min=Math.min(...vals), max=Math.max(...vals), range=max-min||1;

  const W=72, H=22;

  const pts=vals.map((v,i)=>`${(i/(vals.length-1))*W},${H-((v-min)/range)*(H-4)-2}`).join(" ");

  const last=pts.split(" ").pop().split(",");

  return (

    <svg width={W} height={H} style={{display:"block"}}>

      <polyline points={pts} fill="none" stroke={color} strokeWidth="1.5" strokeLinecap="round" strokeLinejoin="round" opacity="0.6"/>

      <circle cx={last[0]} cy={last[1]} r="2.5" fill={color}/>

    </svg>

  );

}
